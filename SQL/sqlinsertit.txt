CREATE DATABASE IF NOT EXISTS Kirpparidb;

CREATE TABLE USER (
UserName varchar (20) PRIMARY KEY,
Password varchar(100),
Email varchar (100),
PhoneNumber int (10),
TypeID int (10),
FOREIGN KEY TypeID REFERENCES USERTYPE(TypeID));


CREATE TABLE USERTYPE (
TypeID int (10) PRIMARY KEY,
Name varchar (30),
RightsLevel int (3)
);

CREATE TABLE PRODUCT (
ProductID int AUTO_INCREMENT PRIMARY KEY,
Name varchar (100),
Description varchar (255),
Price decimal (5,2),
SoldStatus int (1),
UserName varchar (20),
ImageID int (10),
FOREIGN KEY UserName REFERENCES USER(UserName),
FOREIGN KEY ImageID REFERENCES IMAGE(ImageID));

CREATE TABLE IMAGE (
ImageID int (10) PRIMARY KEY,
Title varchar (100),
Location varchar (255),
AlternativeText varchar (255)
)
